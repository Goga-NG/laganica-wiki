<script>
var prefix = window.location.pathname;

function postaviBoju(id, boja) {
  if (boja === '') {
    localStorage.removeItem(prefix + id);
  } else {
    localStorage.setItem(prefix + id, boja);
  }
  primeni();
}

function primeni() {
  var items = document.querySelectorAll('.pitanje-item');
  var zeleno = 0, žuto = 0, crveno = 0, ukupno = 0;
  items.forEach(function(item) {
    var id = item.getAttribute('data-id');
    var boja = localStorage.getItem(prefix + id) || '';
    item.querySelector('.pitanje-link').className = 'pitanje-link ' + boja;
    if (boja === 'zeleno') zeleno++;
    if (boja === 'žuto') žuto++;
    if (boja === 'crveno') crveno++;
    if (boja !== '') ukupno++;
  });
  document.getElementById('zeleno-count').textContent = zeleno;
  document.getElementById('žuto-count').textContent = žuto;
  document.getElementById('crveno-count').textContent = crveno;
  document.getElementById('ukupno-count').textContent = ukupno;
}

document.addEventListener('DOMContentLoaded', primeni);
</script>
